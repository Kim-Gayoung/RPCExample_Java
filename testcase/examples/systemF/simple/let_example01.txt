writeConsole: String-c->Unit = lam^c x: String. #primWriteConsole (x);
readConsole: Unit-c->String = lam^c x: Unit. #primReadConsole (x);
toInt_client: String-c->Int = lam^c x: String. #primToInt_client (x);

height: Int = let a: Unit = writeConsole ^c^ "Enter the height: " in
						toInt_client ^c^ (readConsole ^c^ ())
					end;

width: Int = let a: Unit = writeConsole ^c^ "Enter the width: " in
						toInt_client ^c^ (readConsole ^c^ ())
					end;
					
main: Int = width * height

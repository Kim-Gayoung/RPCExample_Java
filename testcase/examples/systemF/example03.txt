simpleCond: Int-c->(Int-c->String) =
	lam^c x: Int.
	  lam^c y: Int.
		if x >= y then
		  "x GEQ y"
		else
		  "x LT y";
		  
test1: String = simpleCond [Int] [Int] ^c^ 123 ^c^ 456;
test2: String = simpleCond [Int] [Int] ^c^ 987 ^c^ 543;
test3: Int-s->(Int-s->String) =
	lam^s x: Int.
	  lam^s y: Int. simpleCond [Int] [Int] ^c^ x ^c^ y;
test4: Unit-s->String = lam^s x: Unit. test3 [Int] [Int] ^s^ 111 ^s^ 222;

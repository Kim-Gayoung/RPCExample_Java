simpleCond: Bool-c->String = lam^c b: Bool.
  if b == True then
    "This value is True"
  else
  	"This value is False"
;

main: String = let test1: String = simpleCond ^c^ True in
		let test2: Unit-s->String = lam^s x: Unit. simpleCond ^c^ False in
			let test3: Bool-s->string = lam^s b: Bool. simpleCond ^c^ b in
				let test4: String = test2 ^s^ () in
					test3 ^s^ True
				end
			end
		end
	end

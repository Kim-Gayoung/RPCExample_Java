simpleCond: Bool-c->String = lam^c b: Bool.
  if b == True then
    "This value is True"
  else
  	"This value is False"
;

test1: String = simpleCond Bool ^c^ True;
test2: Unit-s->String = lam^s x: Unit. simpleCond Bool ^c^ False;
test3: Bool-s->string = lam^s b: Bool. simpleCond Bool ^c^ b;
test4: String = test2 Unit ^s^ ();
test5: String = test3 Bool ^s^ True
